from sklearn.linear_model import LinearRegression

# Instantiate the model
model = LinearRegression()

# Fit the model on training data
model.fit(X_train[['Glucose']], y_train)  # Assuming 'Glucose' is the feature used for training

# Predict on training set
y_pred_train = model.predict(X_train[['Glucose']])

# Predict on test set (if needed for another plot or evaluation)
y_pred_test = model.predict(X_test[['Glucose']])

# Now plot
plt.scatter(X_test['Glucose'], y_test, color='lightcoral', label='Test Set')
plt.plot(X_train['Glucose'], y_pred_train, color='firebrick', label='Model Prediction')
plt.title('Glucose vs Insulin (Test Set)')
plt.xlabel('Glucose')
plt.ylabel('Insulin')
plt.legend(title='Legend', loc='best', facecolor='white')
